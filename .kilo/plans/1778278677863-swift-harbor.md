```python
import litellm
from app.core.llm_config import get_model_name, get_llm_settings

class LLMService:
    def __init__(self):
        self.settings = get_llm_settings()
        self.model = get_model_name()
        self.fallback = self.settings.fallback_model
    
    async def generate(self, prompt: str, system: str = None, 
                       response_model: type[BaseModel] = None) -> Any:
        """Generate text or structured output via LiteLLM."""
        messages = []
        if system:
            messages.append({"role": "system", "content": system})
        messages.append({"role": "user", "content": prompt})
        
        kwargs = {"model": self.model, "messages": messages}
        if response_model:
            kwargs["response_format"] = response_model
        
        try:
            response = await litellm.acompletion(**kwargs)
            return response.choices[0].message.content
        except Exception:
            # Fallback to secondary model
            kwargs["model"] = self.fallback
            response = await litellm.acompletion(**kwargs)
            return response.choices[0].message.content
```

```python
import re

def _extract_latex_text(tex_content: str) -> str:
    """Strip LaTeX commands, extract readable text from .tex files.
    Handles: \\section{}, \\subsection{}, \\begin{itemize}, \\item, \\textbf{}, etc.
    """
    text = tex_content
    # Remove comments
    text = re.sub(r'%.*', '', text)
    # Extract section titles
    text = re.sub(r'\\(sub)?section\*?\{([^}]*)\}', r'\n\2\n', text)
    # Extract itemize items
    text = re.sub(r'\\item\s*', '- ', text)
    # Strip common commands but keep their content
    for cmd in ['textbf', 'textit', 'emph', 'underline', 'text']:
        text = re.sub(rf'\\{cmd}\{{([^}}]*)\}}', r'\1', text)
    # Remove remaining LaTeX commands
    text = re.sub(r'\\[a-zA-Z]+\*?(?:\[[^\]]*\])?(?:\{[^}]*\})*', '', text)
    # Clean up environments
    text = re.sub(r'\\begin\{[^}]*\}', '', text)
    text = re.sub(r'\\end\{[^}]*\}', '', text)
    # Clean up braces and whitespace
    text = text.replace('{', '').replace('}', '')
    text = re.sub(r'\n\s*\n', '\n\n', text)
    return text.strip()
```

```python
    async def run(self, ctx: GraphRunContext[WorkflowState]) -> AnalyzeMatch:
        # Parse resume + job using parsing_service
        ctx.state.resume = await parse_resume(ctx.state.resume_file)
        ctx.state.job = await parse_job(ctx.state.job_text)
        return AnalyzeMatch()

@dataclass
class AnalyzeMatch(BaseNode[WorkflowState]):
    """Step 2: Analyze resume-job match."""
    async def run(self, ctx: GraphRunContext[WorkflowState]) -> GenerateTailored:
        # Compare skills, experience, requirements
        ctx.state.match_analysis = await analyze_match(ctx.state.resume, ctx.state.job)
        return GenerateTailored()

@dataclass
class GenerateTailored(BaseNode[WorkflowState]):
    """Step 3: Generate tailored resume via LLM."""
    async def run(self, ctx: GraphRunContext[WorkflowState]) -> ValidateOutput:
        result = await tailor_agent.run(
            f"Tailor this resume for the job:\n{format_as_xml(ctx.state.resume)}\n{format_as_xml(ctx.state.job)}"
        )
        ctx.state.tailored = result.output
        return ValidateOutput()

@dataclass
class ValidateOutput(BaseNode[WorkflowState, None, TailoredResumeOutput]):
    """Step 4: Validate the output."""
    async def run(self, ctx: GraphRunContext[WorkflowState]) -> End[TailoredResumeOutput]:
        # Pydantic validation is automatic, but add quality checks
        return End(ctx.state.tailored)

# The graph
tailor_graph = Graph(nodes=[ParseInput, AnalyzeMatch, GenerateTailored, ValidateOutput])
```

- **Visualizable**: `tailor_graph.mermaid_code()` generates workflow diagrams

```python
        return {"Error": data["message"]}
```

## Risk Areas
1. **Pydantic AI Graph complexity** ŌĆö graphs add abstraction; keep workflows simple initially, use plain agent calls where graph isn't needed
2. **Prompt quality** ŌĆö parsing accuracy depends on prompt engineering iteration; test with real resume files from `data/resumes/`
3. **Dual scraper maintenance** ŌĆö two backends = two codepaths to test; TinyFish is primary so Crawl4AI is low priority
4. **Async patterns** ŌĆö Gradio + async FastAPI + async Pydantic AI + async LiteLLM need consistent handling; use `nest_asyncio` for Gradio callbacks
5. **TinyFish rate limits** ŌĆö free tier may be limiting for batch scanning; Fetch API is 0 credits so should be fine for career pages
6. **LiteLLM model availability** ŌĆö different providers have different capabilities; structured output support varies by provider (OpenAI/Anthropic best, Gemini improving)
7. **LaTeX parsing edge cases** ŌĆö custom LaTeX commands, complex nested environments may not strip cleanly; fallback to raw text
8. **Test expectations mismatch** ŌĆö `test_resume_models.py` uses `contact_info`/`work_experience`/`proficiency`/`years_of_experience` field names that must be exactly matched
